cmake_minimum_required(VERSION 2.8)
project(Unnaming)
add_compile_options(-Wall)
#For Raspberry Pi
add_compile_options(-Wno-psabi)
add_subdirectory(src/Map)
add_subdirectory(src/Logic)
add_subdirectory(src/Graphic)
add_subdirectory(src/Object)
aux_source_directory(src DIR_SRC)
add_executable(Unnaming.out ${DIR_SRC})
target_link_libraries(Unnaming.out Map)
target_link_libraries(Unnaming.out Logic)
target_link_libraries(Unnaming.out Graphic)
target_link_libraries(Unnaming.out Object)
